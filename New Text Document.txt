    <script>
        // 1. SET YOUR ANNIVERSARY DATE HERE (Year, MonthIndex, Day)
        // Note: January is 0, February is 1, March is 2, etc.
        const anniversaryDate = new Date(2022, 7, 15); // Example: Jan 15, 2024

        function updateCounter() {
            const now = new Date();
            const difference = now - anniversaryDate;
            const days = Math.floor(difference / (1000 * 60 * 60 * 24));
            document.getElementById('days-count').innerText = days;
        }

        updateCounter();

        // Heart Logic
        const colors = ['#ff4d6d', '#ff758f', '#c9184a', '#ffb3c1', '#ffd700'];
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart-particle');
            const size = Math.random() * 20 + 10;
            const color = colors[Math.floor(Math.random() * colors.length)];
            const duration = Math.random() * 5 + 5;
            heart.style.left = Math.random() * 100 + "vw";
            heart.style.bottom = "-5vh";
            heart.style.fontSize = size + "px";
            heart.style.color = color;
            heart.style.filter = `blur(${Math.random() * 2}px)`;
            heart.style.animation = `floatAndSway ${duration}s ease-in-out forwards`;
            heart.innerHTML = '❤';
            document.body.appendChild(heart);
            setTimeout(() => { heart.remove(); }, duration * 1000);
        }
        setInterval(createHeart, 400);
    </script>
